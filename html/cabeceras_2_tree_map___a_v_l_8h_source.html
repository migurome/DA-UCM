196  static size_t altura(Link a) {

201  static void rotaDer(Link & k2) {
202  Link k1 = k2->iz;
203  k2->iz = k1->dr;
204  k1->dr = k2;
205  k2->altura = std::max(altura(k2->iz), altura(k2->dr)) + 1;
206  k1->altura = std::max(altura(k1->iz), altura(k1->dr)) + 1;
207  k2 = k1;
208  }

210  static void rotaIzq(Link & k1) {
211  Link k2 = k1->dr;
212  k1->dr = k2->iz;
213  k2->iz = k1;
214  k1->altura = std::max(altura(k1->iz), altura(k1->dr)) + 1;
215  k2->altura = std::max(altura(k2->iz), altura(k2->dr)) + 1;
216  k1 = k2;
217  }

229  static void reequilibraIzq(Link & a) {
230  if (altura(a->dr) - altura(a->iz) > 1) {
231  if (altura(a->dr->iz) > altura(a->dr->dr))
232  rotaDerIzq(a);
233  else rotaIzq(a);
234  }
235  else a->altura = std::max(altura(a->iz), altura(a->dr)) + 1;
236  }

238  static void reequilibraDer(Link & a) {
239  if (altura(a->iz) - altura(a->dr) > 1) {
240  if (altura(a->iz->dr) > altura(a->iz->iz))
241  rotaIzqDer(a);
242  else rotaDer(a);
243  }
244  else a->altura = std::max(altura(a->iz), altura(a->dr)) + 1;
245  }

247  void borra(Clave const& c, Link & a) {
248  if (a != nullptr) {
249  if (menor(c, a->cv.clave)) {
250  borra(c, a->iz);
251  reequilibraIzq(a);
252  }
253  else if (menor(a->cv.clave, c)) {
254  borra(c, a->dr);
255  reequilibraDer(a);
256  }
257  else // c == a->cv.clave
258  if (a->iz == nullptr || a->dr == nullptr) {
259  Link aux = a;
260  a = a->iz == nullptr ? a->dr : a->iz;
261  --nelems;
262  delete aux;
263  }
264  else { // tiene dos hijos
265  subirMenor(a, a->dr, nullptr);
266  --nelems;
267  }
268  }

271  static void subirMenor(Link & a, Link b, Link padre) {
272  if (b->iz != nullptr){
273  subirMenor(a, b->iz, b);
274  reequilibraIzq(b);
275  }
276  else {
277  if (padre != nullptr) {
278  padre->iz = b->dr;
279  b->dr = a->dr;
280  }
281  b->iz = a->iz;
282  delete a;
283  a = b;

285  }

338  // construye el iterador a una clave concreta (para find)
339  Iterador(map_t const* m, Clave const& c) {
340  act = m->raiz;
341  bool encontrado = false;
342  while (act != nullptr && !encontrado) {
343  if (m->menor(c, act->cv.clave)) {
344  ancestros.push(act);
345  act = act->iz;
346  } else if (m->menor(act->cv.clave, c)) {
347  act = act->dr;
348  } else
349  encontrado = true;
350  }
351  if (!encontrado) { // vaciamos la pila
352  ancestros = std::stack<Link>();
353  }
354  }
355 
356  Link first(Link ptr) {
357  if (ptr == nullptr) {
358  return nullptr;
359  } else { // buscamos el nodo más a la izquierda
360  while (ptr->iz != nullptr) {
361  ancestros.push(ptr);
362  ptr = ptr->iz;
363  }
364  return ptr;
365  }
366  }
367 
368  void next() {
369  if (act == nullptr) {
370  throw std::out_of_range("El iterador no puede avanzar");
371  } else if (act->dr != nullptr) { // primero del hijo derecho
372  act = first(act->dr);
373  } else if (ancestros.empty()) { // hemos llegado al final
374  act = nullptr;
375  } else { // podemos retroceder
376  act = ancestros.top();
377  ancestros.pop();
378  }
379  }
380  };
