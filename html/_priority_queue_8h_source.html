111  /* Flota el elemento situado en la posición n del montículo. */
112  void flotar(size_t n) {
113  T elem = std::move(array[n]);
114  size_t hueco = n;
115  while (hueco != 1 && antes(elem, array[hueco / 2])) {
116  array[hueco] = std::move(array[hueco / 2]);
117  hueco /= 2;
118  }
119  array[hueco] = std::move(elem);
120  }
121 
122  /* Hunde el elemento situado en la posición n del montículo. */
123  void hundir(size_t n) {
124  T elem = std::move(array[n]);
125  size_t hueco = n;
126  size_t hijo = 2 * hueco; // hijo izquierdo, si existe
127  while (hijo <= numElems) {
128  // cambiar al hijo derecho si existe y va antes que el izquierdo
129  if (hijo < numElems && antes(array[hijo + 1], array[hijo]))
130  ++hijo;
131  // flotar el hijo si va antes que el elemento hundiéndose
132  if (antes(array[hijo], elem)) {
133  array[hueco] = std::move(array[hijo]);
134  hueco = hijo; hijo = 2 * hueco;
135  }
136  else break;
137  }
138  array[hueco] = std::move(elem);
139  }
140 
141  /* Convierte un vector en un montículo. */
142  void monticulizar() {
143  for (auto i = numElems / 2; i > 0; --i)
144  hundir(i);
145  }
