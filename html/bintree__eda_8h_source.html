104  std::vector<T> levelorder() const {
105  std::vector<T> levels;
106  if (!empty()) {
107  std::queue<Link> pendientes;
108  pendientes.push(raiz);
109  while (!pendientes.empty()) {
110  Link sig = pendientes.front();
111  pendientes.pop();
112  levels.push_back(sig->elem);
113  if (sig->left != nullptr)
114  pendientes.push(sig->left);
115  if (sig->right != nullptr)
116  pendientes.push(sig->right);
117  }
118  }
