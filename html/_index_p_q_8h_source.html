119  void flotar(size_t n) {
120  Par parmov = std::move(array[n]);
121  size_t hueco = n;
122  while (hueco != 1 && antes(parmov.prioridad, array[hueco/2].prioridad)) {
123  array[hueco] = std::move(array[hueco/2]); posiciones[array[hueco].elem] = hueco;
124  hueco /= 2;
125  }
126  array[hueco] = std::move(parmov); posiciones[array[hueco].elem] = hueco;
127  }

129  void hundir(size_t n) {
130  Par parmov = std::move(array[n]);
131  size_t hueco = n;
132  size_t hijo = 2*hueco; // hijo izquierdo, si existe
133  while (hijo <= numElems) {
134  // cambiar al hijo derecho de i si existe y va antes que el izquierdo
135  if (hijo < numElems && antes(array[hijo + 1].prioridad, array[hijo].prioridad))
136  ++hijo;
137  // flotar el hijo si va antes que el elemento hundiéndose
138  if (antes(array[hijo].prioridad, parmov.prioridad)) {
139  array[hueco] = std::move(array[hijo]); posiciones[array[hueco].elem] = hueco;
140  hueco = hijo; hijo = 2*hueco;
141  }
142  else break;
143  }
144  array[hueco] = std::move(parmov); posiciones[array[hueco].elem] = hueco;
145  }
